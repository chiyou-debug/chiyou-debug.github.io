    # 4. go语言特点

    • 没有头文件的概念，没有.h文件，只有.go
    • 强类型的语言，编译型的，不是解析行的（python，shell）
    • 一个go语言程序在编译的时候，会把执行所需要的库全部打包到执行程序中（不依赖动态库，静态库）
      • go程序比较大
      • 在执行时非常灵活
      • 编译的时候，如果发现import的包没有被引用，那么编译会出错（为了尽量小一点）
    • go语法不区分平台的，在windows下面编译的程序，也可以在linux下运行，需要两个环境变量来控制：
      • GOOS: 设定运行的平台
        • mac : GOOS=darwin
        • linux: GOOS=linux
        • ==window: GOOS=windows==
      • GOARCH: 设定平台的体系架构
        • 386: GOARCH=386
        • ==amd64: GOARCH=amd64==
        • arm: GOARCH=arm

    # 6. 定长数组

    package main

import "fmt"

func main() {
	//1-定义一个10个数字数组
	//C++: int nums[10] = {1,2,3}
	//go语法：
	nums := [10]int{1, 2, 3, 4} //常用方式
	//var nums [10]int //常用
	//var nums = [10]int{1,2,3,4} //不常用

	//2-遍历数组，方式一: 数组下标法
	for i := 0; i < len(nums); i++ {
		fmt.Println("i:", i, ", j:", nums[i])
	}

	//方式二: for range ==> python支持
	//使用迭代器方式进行变量，value是nums对应元素的副本，修改value是不会修改nums本身的值
	for key, value := range nums {
		value++
		fmt.Println("key:", key, ", value:", value)
	}

	fmt.Println("尝试修改了数组元素的值，看看能否生效!")

	for key, value := range nums {
		fmt.Println("key:", key, ", value:", value)
	}

	//在go语言中，如果一个变量的值你不想使用，不想接收，可以使用下划线进行忽略
	for _, value := range nums {
		fmt.Println("value:", value)
	}

	//讲函数的时候还会涉及到_
}

    # 7. 不定长数组（切片、slice）

    package main

import "fmt"

func main() {
	//cities := [10]string{"北京", "上海", "广州", "深圳"}
	cities := []string{"北京", "上海", "广州", "深圳"} //不定长数组，不指定长度

	for key, value := range cities {
		fmt.Println("key:", key, ", value:", value)
	}

	fmt.Println("追加元素前的长度:", len(cities), "当前容量:", cap(cities))

	fmt.Println("追加元素之后!")
	//01-追加元素，在cities后面追加一个元素，同时赋值回cities
	cities = append(cities, "杭州")

	fmt.Println("追加元素之后的长度:", len(cities), "当前容量:", cap(cities))

	cities = append(cities, "宁波")
	fmt.Println("追加元素之后的长度:", len(cities), "当前容量:", cap(cities))

	for key, value := range cities {
		fmt.Println("key:", key, ", value:", value)
	}

	//使用数字验证一下追加元素时分配空间的规律, 1K以下是按照2倍空间进行重新分配的，这样避免频繁分配效率低
	nums := []int{}
	for i := 0; i < 50; i++ {
		nums = append(nums, i)
		fmt.Println("len:", len(nums), ",cap:", cap(nums))
	}
}

    小结：

    1. 可以使用append进行追加数据
    2. len获取长度，cap获取容量
    3. 如果容量不足，再次追加时，会动态分配2倍空间

    # 7. 切片

    package main

import "fmt"

func main() {
	//基于原有数组，创建新的不定长数组的方式，就切片
	cities := [6]string{"北京", "上海", "广州", "深圳", "厦门", "重庆"}

	//原来的方式（不推荐的）
	cities1 := [2]string{}
	cities1[0] = cities[0]
	cities1[1] = cities[1]

	//使用切片来快速获取,
	// 1. 左闭右开原则
	// 2. 获取的切片是动态数组, 可以追加新的元素
	// 3. 如果从最左端获取数据,那么冒号左边的数字可省略,如果想获取到最后一个元素,冒号右面的数字可省略
	cities2 := cities[1:3] //左闭右开
	fmt.Println("cities2:", cities2)

	//获取后三个
	cities3 := cities[3:6] //左闭右开
	fmt.Println("cities3:", cities3)

	cities3 = append(cities3, "甘肃")
	fmt.Println("cities3:", cities3)

//如果从最左端获取数据， 那么冒号左边的数字可以省略， 如果想获取到最后一个元素， 冒号右面的数字可以省略
	cities4 := cities[:4]
	cities5 := cities[1:]
	fmt.Println("cities4:", cities4)
	fmt.Println("cities5:", cities5)

	//基于cities产生一个全新的切片(把一个定长数组变成切片的方法）
	cities6 := cities[:]
	fmt.Println("cities6:", cities6)

	str1 := "helloworld"
	sub1 := str1[:5]
	fmt.Println("sub1:", sub1)

	//注意，切片的底层还是数组， 相当于引用， 修改切片， 会影响到原始数组 ==> 浅拷贝
	cities7 := cities[1:4] //上海， 广州， 深圳
	fmt.Println("修改之前!")
	fmt.Println("cities:", cities)
	fmt.Println("cities7:", cities7)
	cities7[0] = "黑龙江"

	fmt.Println("修改之后!")
	fmt.Println("cities:", cities)
	fmt.Println("cities7:", cities7)

	//使用copy函数， 进行深拷贝，防止影响原始数组
	//1. 先分配空间(3个空间）, 使用make函数对切片进行空间分配
	cities8 := make([]string, 3, 20)//创建一个string类型的切片,元素个数是3个,默认值都是空,容量是20
	fmt.Println("cites8: len:", len(cities8), ", cap:", cap(cities8))
	//2. 执行copy动作, 这是深拷贝
	copy(cities8, cities[1:4])
	cities8[0] = "堪培拉"
	fmt.Println("cities:", cities)
	fmt.Println("cities8:", cities8)
}

    小结：

    1. 不定长数组就是切片，有len和cap的概念

    2. 使用make来进行分配空间，

      1. make([]string, 长度, 容量)
      2. make([]string, 长度) ， 容量可以省略
      3. 默认与长度相同，空间默认值是零值（0, 空字符串， false，nil）

    3. 切片的底层是数组， 是浅拷贝，深拷贝需要使用copy实现

    map必须分配空间之后才能使用， 使用make进行空间分配

    package main

import "fmt"

func main() {
	//1. 定义一个字典
	//map[类型1]类型2
	var idNames map[int]string //此时的map是空的nil，不能直接赋值

	//idNames = make(map[int]string) //没有指定容量，但是已经不是nil了，可以赋值了，动态扩容
	idNames = make(map[int]string, 10) //有10个容量，超过了再扩容

	idNames[0] = "Duke"
	idNames[1] = "Lily"
	idNames[2] = "Lily1"
	idNames[3] = "Lily2"
	idNames[4] = "Lily3"

	fmt.Println(idNames)

	// 2. 定义的时候直接分配空间
	namesScore := make(map[string]int, 20)
	namesScore["Duke"] = 130
	fmt.Println(namesScore)

	//3. 遍历map
	for key, value := range idNames {
		fmt.Println("key:", key, ", value:", value)
	}

	//4. 访问一个不存在的key，是不会报错的，只会返回这个类型的0值
	fmt.Println("张三的分值:", namesScore["张三"])     //这句代码会报错吗?
	fmt.Println("idName[1000]:", idNames[1000]) //这句代码会报错吗?

	//5. 如何判断一个map是否存在某个key
	value, ok := idNames[1000] //如果1000这个key存在，则ok是true，value是对应的值，否则ok是false，value是零值
	if !ok {
		fmt.Println("1000这个key不存在!")
		//return
	} else {
		fmt.Println("1000这个key存在!")
		fmt.Println("value:", value)
	}

	//6. 如何删除一个map中的key
	fmt.Println("删除之前:", idNames)
	delete(idNames, 0)
	fmt.Println("删除之后:", idNames)
	delete(idNames, 100000)
	fmt.Println("删除之后:", idNames)

	//7. 并发相关的知识, 多个线程对统一map同时读写时，需要上锁，否则会出现资源竞争
}

    package main

import (
	"calculator/add"             //工程目录开始，到指定的包名
	. "calculator/div"           //使用. , 之后调用时不需要指定包名，直接写函数名即可，不推荐
	mymultiple "calculator/mult" //导入之后，修改包的名字
	"calculator/sub"
	"fmt"
)

//一个程序，必须要一个package main
//包名.函数形式
//如果在包外不调用，需要写成大写字母开头

func main() {
	res := add.Add(10, 20)
	fmt.Println("add res:", res)

	res = sub.Sub(20, 10)
	fmt.Println("sub res:", res)

	res = mymultiple.Mult(10, 20)
	fmt.Println("mymultiple res:", res)

	res = Div(100, 20)
	fmt.Println("Div res:", res)
}
