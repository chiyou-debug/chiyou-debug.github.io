        # 1. 文件操作

        • QFile

          fopen();
// 打开文件 -> 如果操作非utf8格式文件中文会是乱码
QFile::QFile(const QString &name);
QFile::QFile(const QString &name, QObject *parent);
参数:
	name: 要操作的磁盘文件

// 打开文件
[virtual] bool QIODevice::open(OpenMode mode);
参数: mode
	- 只读: QIODevice::ReadOnly 0x0001
    - 只写: QIODevice::WriteOnly 0x0002
    - 读写: QIODevice::ReadWrite
    - 追加: QIODevice::Append		-> 写文件
    - 文件截断: QIODevice::Truncate	-> 写文件
返回值: 打开成功: true, 失败: false

// char buf[1024];	string buf1;
// 文件操作(读, 写)
// 读文件
qint64 QIODevice::read(char *data, qint64 maxSize);
QByteArray QIODevice::read(qint64 maxSize);	// 读指定字节数
QByteArray QIODevice::readAll();			// 读文件所有内容
// 读一行
qint64 QIODevice::readLine(char *data, qint64 maxSize);
QByteArray QIODevice::readLine(qint64 maxSize = 0)
// 判读是否读到文件尾部
[virtual] bool QIODevice::atEnd() const;

// 写文件
qint64 QIODevice::write(const char *data, qint64 maxSize);
qint64 QIODevice::write(const char *data);
qint64 QIODevice::write(const QByteArray &byteArray);

// 关闭文件
[virtual] void QIODevice::close();
