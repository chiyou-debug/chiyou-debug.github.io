                # libevent

                # 1. Libevent介绍

                • 特点

                  Libevent 是一个用C语言编写的、轻量级的开源高性能事件的框架，主要有以下几个亮点：

                  • 事件驱动（ event-driven），高性能;
                    • libevent会通过IO转接函数对事件进行检测
                    • 会通过回调函数进行处理
                  • 轻量级，专注于网络；
                  • 源代码相当精炼、易读；
                    • 现阶段我不建议
                  • 跨平台，支持 Windows、 Linux、 BSD（是Unix的衍生系统） 和 Mac OS；
                  • 支持多种 I/O 多路复用技术， epoll、 poll、 select 和 kqueue 等；
                  • 支持 I/O，定时器和信号等事件；
                    • 处理读事件
                    • 处理写事件
                    • 异常事件
                    • 超时处理
                    • 支持linux中的信号
                      • 当进程中产生某个信号的时候, 这个也称之为事件
                  • 支持注册事件优先级。
                    • 默认创建的出的事件优先级相同
                    • 可以通过操作函数设置事件不同的优先级

                # 2. 事件处理框架 - event_base

                使用 libevent函数之前需要分配一个或者多个 event_base 结构体。每个event_base 结构体持有一个事件集合，可以检测以确定哪个事件是激活的。每个 event_base 都有一种用于检测哪种事件已经就绪的 “方法”，或者说后端。

                • API函数

                  // 头文件
#include <event2/event.h>
// 操作函数
struct event_base * event_base_new（void）;	// 要掌握的函数
void event_base_free（struct event_base * base）;

// 检查event_base的后端方法 -> 封装的IO转接函数
const char** event_get_supported_methods(void）;
// 查看当前操作系统使用的什么IO转接函数
const char *event_base_get_method(const struct event_base *base);

                # 3. 事件循环

                event_base不停的检测委托的检测是实际是不是发生了, 如果发生了, event_base会调用对应的回调函数, 这个回调函数的用户委托检测事件的时候给的.

                • 设置事件循环

                  // 头文件
#include <event2/event.h>

// 只要这个函数被调用, 就可以对event_base中的事件进行检测
// 如果检测了一遍: 发现event_base中没有注册任何事件 -> 循环退出
// 发现event_base中有注册事件, 但是事件并没有被触发 -> 继续检测
// 发现event_base中有注册事件, 但是事件被触发 -> 事件被处理, 默认这个事件只处理一次
// 			- 如果想持续检测这个事件, 需要额外设置
int event_base_dispatch(struct event_base* base);	// 一般使用这个函数

                • 终止事件循环

                  // 头文件
#include <event2/event.h>

// 表示一个时间段, tv_sec + tv_usec
struct timeval {
	long    tv_sec;                    
	long    tv_usec;    // 微秒        
};

// 等当前这一轮循环完成之后再结束
// 如果当前没有要处理的事件, 直接终止了
int event_base_loopexit(struct event_base * base, const struct timeval * tv);
	参数:
		- base: 通过event_base_new（void）得到的事件处理框架
		- tv: 在这个时间段之后退出循环
// 事件循环马上结束
int event_base_loopbreak(struct event_base * base);

                # 4. 事件

                • 事件的创建、添加、删除

                  // 头文件
#include <event2/event.h>

#define EV_TIMEOUT 	0x01	// 超时
#define EV_READ 	0x02	// 读事件
#define EV_WRITE 	0x04	// 写事件
#define EV_SIGNAL 	0x08	// 信号事件
#define EV_PERSIST 	0x10	// 设置事件被持续检测
#define EV_ET 		0x20	// 设置边沿模式

// 事件处理函数的函数原型
typedef void (*event_callback_fn)(evutil_socket_t fd, short what, void *arg);
	参数:
		- fd: 文件描述符, 是event_new函的第二个参数被传递进来了
		- what: 存储了实际被触发的事件
		- arg: event_new函数的第五个参数
		
// 创建一个事件
// evutil_socket_t == int
struct event* event_new(struct event_base * base，evutil_socket_t fd，
     					 short what，event_callback_fn cb，void * arg);
	参数:
		- base: 事件的处理框架
		- fd: 一个文件描述符
			- open();
			- socket();
			- pipe();
		- what: 要检测第二个参数fd的什么事件, 参考上边的宏定义
			EV_TIMEOUT | EV_READ | EV_WRITE
		- cb: 事件产生之后的处理动作
		- arg: 作为参数传递给回调函数cb

// 释放事件资源, 参数是event_new的返回值
void event_free(struct event * event);

// 添加事件
struct timeval {
	long    tv_sec;                    
	long    tv_usec;    // 微秒        
};
// 将创建出的新的事件添加到event_base事件处理框架上
// 这样在事件循环的时候才能检测到这个事件
int  event_add(struct event * ev，const  struct timeval * tv);
	参数:
		- ev: 创建的新事件
		- tv: 超时时长, 假设在设置的时间段之内检测的事件没有触发
			- 如果超时了, 强迫事件的处理函数被调用

// 将事件从event_base事件处理框架上删除, 这个事件就不会被检测了
int  event_del(struct event * ev);


                  /*
	1. 创建事件处理框架 event_base
	2. 创建新的事件 event, 这时候还不能被 event_base 处理
	3. 将事件添加到 event_base 上
	4. event_base开始事件循环 event_base_dispatch
		- 事件触发, 事件被处理 -> 这个事件从 event_base 移除了
			- 如果想要被持续处理
				- 方式1: 在添加一次 event_add()
				- 方式2: event_new, 参数what指定 EV_PERSIST   -> 推荐
*/

                • 事件的优先级设置

                  // 头文件
#include <event2/event.h>
// EVENT_MAX_PRIORITIES == 256
// 初始化事件优先级有多少级
int event_base_priority_init(struct event_base * base，int n_priorities);
	参数:

	参数:

		- priority: 按照 event_base_priority_init() 指定的等级范围指定就行

                # 5. 带缓冲区的事件

                // 网络通信的时候要使用文件描述符
// 文件描述符在内核中对应两块内存
	- 读
	- 写
// 带缓冲区的事件 -> 叫做: bufferevent
// 	- libevent又额外提供了一块内存, 是对内核提供的缓冲区的封装
// 		- 我们操作的libevent提供的缓冲区
			- 读缓冲区 -> 数据来自于内核的读缓冲区
			- 写缓冲区 -> 数据流向内核的写缓冲区


                // bufferevent提供了两块缓冲区
	- 当读缓冲区中有数据, 读事件被触发, 读缓冲区对应的回调会被调用
	- 写缓冲区也对应一个回调函数
		- 这个函数没什么用, 为什么没有用?
        	- 使用者将数据写入写缓冲区, 这时候回调函数会被调用
        		通知使用者, 数据成功被写入到了内核中

                • 概念理解

                  1. bufferevent 理解:
                    • 是libevent为IO缓冲区操作提供的一种通用机制
                    • bufferevent 由一个底层的传输端口(如套接字 ), 一个读取缓冲区和一个写入缓冲区组成。
                    • 与通常的事件在底层传输端口已经就绪, 可以读取或者写入的时候执行回调不同的是, bufferevent在读取或者写入了足够量的数据之后调用用户提供的回调。
                  2. 回调 - 缓冲区对应的操作
                  3. 每个 bufferevent 有两个数据相关的回调
                  • 读取回调: 从底层传输端口读取了任意量的数据之后会调用读取回调(默认)
                    • 当libevent维护的读缓冲区有数据的时候(从内核缓冲区中读过来), 调用缓冲区的读回调
                    • 数据量到达多少多少读回调被调用, 这是可以设置的
                  • 写入回调: 输出缓冲区中足够量的数据被清空到底层传输端口后写入回调会被调用(默认)
                    • 当数据被写入到写缓冲区中之后, 该函数被调用
                    • 这个回调函没什么用

                • 创建/释放基于套接字的bufferevent

                  // 创建带缓冲区的事件
// 可以将一个现有的文件描述符进行包装
struct bufferevent *bufferevent_socket_new(
	struct event_base *base,
	evutil_socket_t fd,
	enum bufferevent_options options
); 
参数:
	- base: 事件处理框架
	- fd: 要包装的文件描述符
		- 情况1: 指定一个有效的文件描述符
		- 情况2: -1, 没有文件描述符, 让当前函数在内部创建一个有效的文件描述符
	- options: 选项
		- BEV_OPT_CLOSE_ON_FREE: 当前事件释放, 并且关闭封装的文件描述符

// 释放资源
void bufferevent_free(struct bufferevent *bev);

                • 在bufferevent上启动链接

                  // 连接服务器函数
int bufferevent_socket_connect(
	struct bufferevent *bev,
	struct sockaddr *address,
	int addrlen
); 
参数:
	- bev: 带缓冲区的事件, 里边封装了用于通信的文件描述符
	- address: 里边要初始化服务器端的地址信息
	- addrlen: 参数address占用的内存大小

                • bufferevent读写缓冲区回调操作

                  typedef void (*bufferevent_data_cb)(struct bufferevent *bev, void *ctx);
	参数:
		- bev: bufferevent_setcb()函数的第一个参数被作为实参传递给了形参bev
		- ctx: bufferevent_setcb()函数的最后一个参数作为实参传递给了形参ctx
		
typedef void (*bufferevent_event_cb)(struct bufferevent *bev, short events, void *ctx);
	参数:
		- bev: bufferevent_setcb()函数的第一个参数被作为实参传递给了形参bev
		- events: 可以检测到的事件, 检测方式: events & BEV_EVENT_CONNECTED
			EV_EVENT_READING：读取操作时发生某事件，具体是哪种事件请看其他标志。
				- 读操作产生了异常, 举例: 读缓冲区没数据并且做了读操作(读不阻塞)
			BEV_EVENT_WRITING：写入操作时发生某事件，具体是哪种事件请看其他标志。
				- 写操作产生了异常, 举例: 写缓冲区满了并且做了写操作(写不阻塞的情况下)
			BEV_EVENT_ERROR：操作时发生错误。关于错误的更多信息
				- 请调用 EVUTIL_SOCKET_ERROR()。
			BEV_EVENT_TIMEOUT：发生超时。
			BEV_EVENT_EOF：遇到文件结束指示。
				- 通信的另外一端已经关闭了连接
			BEV_EVENT_CONNECTED：请求的连接过程已经完成 
				- 这个判定是在客户端做的
		- ctx: bufferevent_setcb() 函数中的最后一个参数 cbarg, 被传递到该函数中	

// 给读写缓冲区设置回调函数
void bufferevent_setcb(struct bufferevent *bufev, 
       					bufferevent_data_cb readcb, 		
       					bufferevent_data_cb writecb, 
                       	bufferevent_event_cb eventcb, void *cbarg
);
	参数:
		- bufev: 带缓冲区的事件, 里边封装了用于通信的文件描述符
		- readcb: 读缓冲区对应的回调函数
			- 调用的时机: 当读缓冲区中有数据的时候, 这个函数被自动调用
		- writecb: 写缓冲区对应的回调函数
			- 调用的时机: 当数据被写入到写缓冲区之后, 这个函数被自动调用 -> 鸡肋
		- eventcb: 事件对应的回调 (读写不算)
            - 异常
            - 连接成功
            - 断开连接
            - 信号事件
         - cbarg: 会被传递到回调函数中

                • 禁用、启用缓冲区

                  /*
	可以启用或者禁用 bufferevent 上的 EV_READ、EV_WRITE 或者 EV_READ | EV_WRITE 事件。
	没有启用读取或者写入事件时, bufferevent 将不会试图进行数据读取或者写入。
*/
// 写缓冲区默认是可用的, 读缓冲区默认不可用
// 写缓冲区默认是可用的 -> 写回调在适当的时机就会被调用
// 读缓冲区默认不可用 -> 读回调函数不会被调用
void bufferevent_enable(struct bufferevent *bufev, short events); 
	- bufev: 带缓冲区的事件, 里边封装了用于通信的文件描述符
	- events: 缓冲区的事件
		- EV_READ
		- EV_WRITE

// 设置某个事件无效
void bufferevent_disable(struct bufferevent *bufev, short events);

// 获取缓冲区对应的有效事件
short bufferevent_get_enabled(struct bufferevent *bufev); 
返回值: 得到的缓冲区的状态, 比如返回值为 flag
	- 检测是不是可读: flag & EV_READ > 0(EV_READ)
    - 检测是不是可写: flag & EV_WRITE > 0(EV_WRITE)

                • 操作bufferevent中的数据

                  // 向bufferevent的输出缓冲区添加数据
int bufferevent_write(struct bufferevent *bufev, const void *data, size_t size);
	参数:
		- bufev: 带缓冲区的事件, 里边封装了用于通信的文件描述符
		- data: 要发送的数据
  		- size: 要发送的数据的长度
		
// 将数据从读缓冲区读出
size_t bufferevent_read(struct bufferevent *bufev, void *data, size_t size);
	参数:
		- bufev: 带缓冲区的事件, 里边封装了用于通信的文件描述符

                # udp

                1. udp 的通信流程
                  • 创建通信的套接字
                    • sock_dgram
                  • 如果这进程已启动之后就需要接收数据
                    • 绑定本地的IP和端口
                  • 通信
                    • recvfrom
                    • sendto
                  • 关闭套接字
                    • close();
                2. 两种模式
                  • 广播
                    • 局域网
                    • 需要广播地址 -> xxx.xxx.xxx.255
                    • 如何发送广播?
                      • 创建通信的udp套接字
                      • 设置广播属性
                        • setsockopt();
                      • 开始广播, 发送数据, 需要使用一个地址(接收数据端地址)
                        • 接收数据端地址
                          • IP: 广播地址
                          • port: 随便找一个没有被使用的端口进行广播
                        • sendto();
                        • 关闭套接字
                    • 如何接收广播?
                      • 创建通信的udp套接字
                      • 绑定端口, 这个端口和发送广播使用的端口相同
                      • 接收数据
                      • 关闭套接字
                  • 组播
                    • 局域网, 广域网都可以使用
                    • 需要组播地址, 需要将数据发送到某个端口
                    • 如何发送组播消息?
                      • 创建通信的udp套接字
                      • 设置组播属性
                        • setsockopt();
                      • 发送数据, 需要使用一个地址(接收数据端地址)
                        • 接收数据端地址
                          • IP: 组播地址
                          • 端口: 随便找一个没有被使用的端口进行组播
                      • 关闭套接字
                    • 如何接收数据?
                      • 创建通信的udp套接字
                      • 绑定端口, 这个端口就是组播的一端使用的端口对应
                      • 加入到多播组
                        • setsockopt();
                      • 接收数据
                      • 关闭套接字
                  • 本地套接字
                    • 使用的通信流程和tcp通信一样
