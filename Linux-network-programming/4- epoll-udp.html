    # 1. select和poll

    # 1.1 select

    # 1.2 poll

    • poll和select的区别

      • select 最多只能检测 1024个文件描述符

        • fd_set 大小是在内核中写死的, 如果要改需要改内核源码, 然后重新编译

        poll 没有 1024的限制，硬件配置越高，支持检测的个数就越多

      • select使用一个标志位来表示一个连接

        poll使用的一个结构体来表示一个连接

      • select可以跨平台, poll只能在linux下使用

    • 相同点

      • 内核在检测poll和select指定的文件描述符的时候, 都是以线性的方式来检测的, 当检测的节点很多的时候, 效率是非常低的.
      • 效率高的情况:
        • 检测的文件描述符不多, 并且这些文件描述符大部分都是活跃的
          • 活跃的: 一直在通信的文件描述符

    • 函数

      #include <poll.h>
struct pollfd {
	int   fd;         /* 委托内核检测的文件描述符 */
	short events;     /* 委托内核检测文件描述符的什么事件 */
	short revents;    /* 文件描述符实际发生的事件 */
};
事件:
	- POLLIN: 读事件
	- POLLOUT: 写事件
	- POLLERR: 异常事件
	
struct pollfd fd;
委托内核检测状态:
	- 检测读缓冲区状态:
		fd.fd = 3;
		fd.events = POLLIN;
	- 检测写缓冲区状态:
		fd.fd = 3;
		fd.events = POLLOUT;
	- 检测读/写缓冲区状态:
		fd.fd = 3;
		fd.events = POLLIN | POLLOUT;
	- 检测所有事件:
		fd.fd = 3;
		fd.events = POLLIN | POLLOUT | POLLERR;


int poll(struct pollfd *fds, nfds_t nfds, int timeout);
	参数:
		- fds: struct pollfd 类型的数组的地址, 里边存储了要委托内核检测的所有的文件描述符
		- nfds: 数组的元素个数, 通过这个大小遍历数组fds
		- timeout: 函数阻塞的时长, 单位是毫秒 ms
			- 0: 不阻塞,直接返回
			- -1: 一直阻塞, 直到检测的文件描述符有状态变化, 解除阻塞
			- >0: 阻塞的毫秒数, 
				- 在阻塞期间如果检测的文件描述符有状态变化, 解除阻塞
				- 没有变化, 时间到了解除阻塞
	返回值:
		-1: 失败
		>0: 成功, 发送变化的文件描述符的个数

      示例

      #include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <arpa/inet.h>
#include <sys/select.h>
#include <poll.h>

int main()
{
    // 1.创建套接字
    int lfd = socket(AF_INET, SOCK_STREAM, 0);
    if(lfd == -1)
    {
        perror("socket");
        exit(0);
    }
    // 2. 绑定 ip, port
    struct sockaddr_in addr;
    addr.sin_port = htons(10000);
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    int ret = bind(lfd, (struct sockaddr*)&addr, sizeof(addr));
    if(ret == -1)
    {
        perror("bind");
        exit(0);
    }
    // 3. 监听
    ret = listen(lfd, 100);
    if(ret == -1)
    {
        perror("listen");
        exit(0);
    }
    
    // 4. 等待连接 -> 循环
    // 检测 -> 读缓冲区, 委托内核去处理
    // 数据初始化, 创建自定义的文件描述符集
    struct pollfd fds[1024];
    // 初始化
    for(int i=0; i<1024; ++i)
    {
        fds[i].fd = -1;
        fds[i].events = POLLIN;
    }
    fds[0].fd = lfd;

    int maxfd = 0;
    while(1)
    {
        // 委托内核检测
        ret = poll(fds, maxfd+1, -1);
        if(ret == -1)
        {
            perror("select");
            exit(0);
        }

        // 检测的读缓冲区有变化
        // 有新连接
        if(fds[0].revents & POLLIN)
        {
            // 接收连接请求
            struct sockaddr_in sockcli;
            int len = sizeof(sockcli);
            // 这个accept是不会阻塞的
            int connfd = accept(lfd, (struct sockaddr*)&sockcli, &len);
            // 委托内核检测connfd的读缓冲区
            int i;
            for(i=0; i<1024; ++i)
            {
                if(fds[i].fd == -1)
                {
                    fds[i].fd = connfd;
                    break;
                }
            }
            maxfd = i > maxfd ? i : maxfd;
        }
        // 通信, 有客户端发送数据过来
        for(int i=1; i<=maxfd; ++i)
        {
            // 如果在集合中, 说明读缓冲区有数据
            if(fds[i].revents & POLLIN)
            {
                char buf[128];
                int ret = read(fds[i].fd, buf, sizeof(buf));
                if(ret == -1)
                {
                    perror("read");
                    exit(0);
                }
                else if(ret == 0)
                {
                    printf("对方已经关闭了连接...\n");
                    close(fds[i].fd);
                    fds[i].fd = -1;
                }
                else
                {
                    printf("客户端say: %s\n", buf);
                    write(fds[i].fd, buf, strlen(buf)+1);
                }
            }
        }
    }
    
    close(lfd);

    return 0;
}

    # 2. epoll

    如果内存1G, epoll就支持10万连接

    • 内核的遍历方式:
    • 红黑树
      • 树 -> 二叉树 -> 二叉排序树 -> 平衡二叉树 -> 红黑树 -> 红黑树变种
      • 查找速度很快
    • 支持检测的文件描述符个数 -> 硬件
    • 使用select和poll
      • 使用的数据需要拷贝
        • select的三个集合, poll的结构体
        • 从用户区 -> 内核区
        • 内核区 -> 用户区
      • epoll是不需要拷贝的
        • 使用的共享内存存储数据
          • 用户区和内核区共享

    # 2.1 epoll的使用

    /*
	epoll使用步骤:
		- 创建一个epoll模型 -> 一个函数
		- 将需要检测的文件描述符添加到epoll模型中
		- 开始使用epoll检测文件描述符的状态
		- 检测到了状态变化
			- 判断是不是监听的fd
				- accept接受连接, 得到新的通信fd, 将其添加到epoll树上
			- 判断是不是通信的fd
				- read
					- 返回值为0 -> 对方关闭连接
						- 将这个fd从树上删除
				- write
*/

    • 函数

      // 创建一个epoll模型 
#include <sys/epoll.h>
int epoll_create(int size);
	参数:
		- size: 保证其大于0即可, 没有其他的实际意义
	返回值:
		是一个文件描述符, 理解为创建的epoll树的根节点

// 对epoll树操作函数, 添加节点/删除节点/修改节点属性
typedef union epoll_data {
	void        *ptr;
	int          fd;	// 通常使用这个变量, 和epoll_ctl第三个参数值相同即可
	uint32_t     u32;
	uint64_t     u64;
} epoll_data_t;

struct epoll_event {
	uint32_t     events;      /* Epoll events */
	epoll_data_t data;        /* User data variable */
};

events事件的取值:
	- EPOLLIN: 读事件  -> 检测是fd的读缓冲区
	- EPOLLOUT: 写事件 -> 检测是fd的写缓冲区
	- EPOLLERR: 异常   -> 检测fd是否有异常

int epoll_ctl(int epfd, int op, int fd, struct epoll_event *event);
	参数:
		- epfd: epoll数的根节点, 通过 epoll_create(int size); 得到的返回值
		- op: 对文件描述符的操作
			EPOLL_CTL_ADD: 添加新节点
			EPOLL_CTL_MOD: 修改节点的属性
			EPOLL_CTL_DEL: 删除现有的节点
		- fd: 要操作的文件描述符
		- event: fd的属性设置
			- 如果是删除操作, 这个参数指定为NULL即可
			
// 委托内核检测epoll树上的文件描述符状态
int epoll_wait(int epfd, struct epoll_event *events, int maxevents, int timeout);
	参数:
		- epfd: epoll数的根节点, 通过 epoll_create(int size); 得到的, 返回值
		- events: 传出参数, 指向一个数组的地址, 存储发生变化的文件描述符的信息
		- maxevents: 参数events元素的个数
		- timeout: 阻塞的时长, 单位是毫秒, ms
			- 0: 直接返回
			- -1: 一直阻塞, 直到检测的文件描述符有状态变化, 解除阻塞
			- >0: 阻塞的毫秒数, 
				- 在阻塞期间如果检测的文件描述符有状态变化, 解除阻塞
				- 没有变化, 时间到了解除阻塞
	返回值:
		-1: 失败
		>0: 状态有变化的文件描述符的个数

    /*
	epoll使用步骤:
		- 创建一个epoll模型 -> 一个函数
		- 将需要检测的文件描述符添加到epoll模型中
		- 开始使用epoll检测文件描述符的状态
		- 检测到了状态变化
			- 判断是不是监听的fd
				- accept接受连接, 得到新的通信fd, 将其添加到epoll树上
			- 负判断是不是通信的fd
				- read
					- 返回值为0 -> 对方关闭连接
						- 将这个fd从树上删除
				- write
*/
int main()
{
    // 1. 创建监听的套接字
    int lfd = socket();
    // 2. lfd和本地的IP和端口绑定
    bind();
    // 3. 设置监听
    listen();
    // 4. 创建epoll模型
    int epfd = epoll_create(100);
    // 5. 初始化epoll模型
    // 将监听的lfd添加的树上
    struct epoll_event ev;
    ev.events = EPOLLIN;
    ev.data.fd = lfd;
    epoll_ctl(epfd, EPOLL_CTL_ADD, lfd, &ev);
    // 6. 检测
    struct epoll_event evs[1024];
    while(1)
    {
    	int num = epoll_wait(epfd, evs, 1024, -1);
        for(int i=0; i<num; ++i)
        {
            int curfd = evs[i].data.fd;
            // 1. 监听的fd
            if(curfd == lfd)
            {
                int cfd = accept();
                ev.events = EPOLLIN;
    			ev.data.fd = cfd;
                epoll_ctl(epfd, EPOLL_CTL_ADD, cfd, &ev);
            }
            // 2. 通信的fd
            else
            {
                int len = read();
                if(len == 0)
                {
                    printf("client disconnect...\n");
                    epoll_ctl(epfd, EPOLL_CTL_DEL, curfd, NULL);
                    close(curfd);
                }
                write();
            }
        }
    }

    return 0;
}

    示例

    // epoll.c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <arpa/inet.h>
#include <sys/epoll.h>

int main()
{
    // 1. 创建监听的套接字
    int lfd = socket(AF_INET, SOCK_STREAM, 0);
    if(lfd == -1)
    {
        perror("socket");
        exit(0);
    }
    // 2. lfd绑定本地的IP和端口
    struct sockaddr_in addr;
    addr.sin_family = AF_INET;
    addr.sin_port = htons(9999);   // 使用大端的端口
    addr.sin_addr.s_addr = INADDR_ANY;  // 这个宏的值就是0
    int ret = bind(lfd, (struct sockaddr*)&addr, sizeof(addr));
    if(ret == -1)
    {
        perror("bind");
        exit(0);
    }
    // 3. 设置监听
    ret = listen(lfd, 128);
    if(ret == -1)
    {
        perror("listen");
        exit(0);
    }

    // 4. 创建epoll树
    int epfd = epoll_create(1);
    if(epfd == -1)
    {
        perror("epoll_create");
        exit(0);
    }

    // 将lfd添加到epollo树
    struct epoll_event ev;
    ev.events = EPOLLIN;
    ev.data.fd = lfd;
    ret = epoll_ctl(epfd, EPOLL_CTL_ADD, lfd, &ev);
    if(ret == -1)
    {
        perror("epoll_ctl");
        exit(0);
    }

    // 检测
    struct epoll_event evs[128];
    int size = sizeof(evs) / sizeof(struct epoll_event);
    while(1)
    {
        int num = epoll_wait(epfd, evs, size, -1);
        printf("epoll_wait return value: %d\n", num);

        for(int i=0; i<num; ++i)
        {
            int curfd = evs[i].data.fd;
            // 判断是不是监听的
            if(curfd == lfd)
            {
                int cfd = accept(lfd, NULL, NULL);
                // 通信的cfd添加到epoll树上
                ev.events = EPOLLIN;
                ev.data.fd = cfd;
                ret = epoll_ctl(epfd, EPOLL_CTL_ADD, cfd, &ev);
                if(ret == -1)
                {
                    perror("epoll_ctl");
                    exit(0);
                }
            }
            // 判断是不是通信的
            else
            {
                char buf[5];
                int len = read(curfd, buf, sizeof(buf));
                if(len == 0)
                {
                    printf("client disconnect\n");
                    // 将当前文件描述符从树上删除
                    epoll_ctl(epfd, EPOLL_CTL_DEL, curfd, NULL);
                    close(curfd);
                }
                else if(len > 0)
                {
                    // 通信
                    write(curfd, buf, len);
                }
                else
                {
                    perror("read");
                    exit(0);
                }
            }
        }
    }

    return 0;
}

    # 2.2 epoll 的工作模式

    // 两种模式:
	- 默认模式: 水平模式 -> LT
		- 处理起来比较简单, 通信效率比较低
		- 套接字属性不需要设置, 使用阻塞模式就可以
	- 需要设置的模式: 边沿模式 -> ET
		- 处理起来比较复杂, 通信效率比较高
		- 套接字必须设置为非阻塞
			- 接收数据的时候, 就需要特殊处理
			
在处理文件描述符对应的缓冲区的时候, 处理方式不太一样
	- 读缓冲区
	- 写缓冲区
    • LT模式

    LT(level triggered)是缺省的工作方式，并且同时支持block和no-block socket。在这种做法中，内核告诉你一个文件描述符是否就绪了，然后你可以对这个就绪的fd进行IO操作。如果你不作任何操作，内核还是会继续通知你的。

    • 读操作 -> 读缓冲区的操作

    • 在程序中调用了 epoll_wait, epoll委托内核检测读缓冲区

      • 从没有数据 -> 有数据
        • 函数调用会返回 ==> 通知程序猿
      • 读缓冲区中的数据被读走了一部分, 还剩一部分, 再次检测的时候, 还有数据
        • 函数调用会返回 ==> 通知程序猿
      • 缓冲区数据被读完了, 再次检测的时候, 没有数据了
        • 函数调用不会返回 ==> 阻塞

    • 写操作 -> 检测写缓冲区是不是可写

      • 写缓冲区有空间 -> 可写

        • epoll_wait函数会一直返回

      • 写缓冲区没有空间 -> 不可写

        • epoll_wait函数阻塞
    /*
    读事件特点:
        epoll检测读缓冲区, 只要有数据, epoll_wait()函数就会返回, 通知使用者
    写事件的特点:
        epoll检测写缓冲区, 只要写缓冲区不满, epoll_wait()函数就会返回, 通知使用者
*/

    • ET模式

      ET(edge-triggered)是高速工作方式，只支持no-block socket。在这种模式下，当描述符从未就绪变为就绪时，内核通过epoll告诉你。然后它会假设你知道文件描述符已经就绪，并且不会再为那个文件描述符发送更多的就绪通知，直到你做了某些操作导致那个文件描述符不再为就绪状态了。但是请注意，如果一直不对这个fd作IO操作(从而导致它再次变成未就绪)，内核不会发送更多的通知(only once)。

      ET模式在很大程度上减少了epoll事件被重复触发的次数，因此效率要比LT模式高。epoll工作在ET模式的时候，必须使用非阻塞套接口，以避免由于一个文件句柄的阻塞读/阻塞写操作把处理多个文件描述符的任务饿死。

      /*
    读事件特点:
        epoll检测读缓冲区, 有数据之后, 只通知一次, 如果数据没有读完, 后续也不会再次通知了
    写事件的特点:
        epoll检测写缓冲区
            - 如果能写: 通知一次
            - 不能写变成能写: 通知一次
*/

      • 读操作 -> 读缓冲区的操作

        • 在程序中调用了 epoll_wait, epoll委托内核检测读缓冲区
          • 从没有数据 -> 有数据
            • 函数调用会返回 ==> 通知程序猿
        • 读缓冲区中的数据被读走了一部分, 还剩一部分, 再次检测的时候, 还有数据
          • 函数调用不会返回 => 阻塞
          • 当读缓冲区有数据之后, 只通知一次, 如果数据没读完, 不会再次通知
        • 缓冲区数据被读完了, 再次检测的时候, 没有数据了
          • 函数调用不会返回 ==> 阻塞

      • 写操作 -> 检测写缓冲区是不是可写

        • 写缓冲区有空间 -> 可写
          • epoll_wait函数会返回，只通知一次
        • 写缓冲区没有空间 -> 不可写
          • epoll_wait函数阻塞
        • 写缓冲区没有空间 -> 有空间
          • epoll_wait函数会返回, 只通知一次

      // 接收数据
ssize_t recvfrom(int sockfd, void *buf, size_t len, int flags,
                        struct sockaddr *src_addr, socklen_t *addrlen);
	参数:
		- sockfd: 通信的文件描述符
		- buf: 接收数据的缓冲区
		- len: 参数buf对应的存储数据的内存大小
		- flags: 文件描述符的属性不需要设置, 指定为0
		- src_addr: 传出参数, 存储了发送数据的一端的地址信息
			- 得到了发送端的IP和端口
		- addrlen: 传入传出参数, 描述的是参数 src_addr对应的内存大小
	返回值:
		失败: -1
        成功: >0
            
// 发送数据
ssize_t sendto(int sockfd, const void *buf, size_t len, int flags,
                      const struct sockaddr *dest_addr, socklen_t addrlen);
	参数:
		- sockfd: 通信的文件描述符
		- buf: 指向的内存中存储了要发送的数据
		- len: 要发送的数据的长度
		- flags: 文件描述符的属性不需要设置, 指定为0
		- dest_addr: 接收数据的一端的地址
		- addrlen: dest_addr指针对应的内存大小
