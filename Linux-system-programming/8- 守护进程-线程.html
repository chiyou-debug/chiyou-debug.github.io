          # 守护进程 - 线程

          # 1. 守护进程

          守护进程（Daemon Process），也就是通常说的 Daemon 进程（精灵进程），是 Linux 中的后台服务进程。它是一个生存期较长的进程，通常独立于控制终端并且周期性地执行某种任务或等待处理某些发生的事件。一般采用以d结尾的名字。

          # 1.1 守护进程的特点

          • 名字的命名: 一般情况下守护进程的名字以d为后缀 xxxxd

          • 守护进程不依赖任何终端, 进程变成守护进程之后就脱离了终端

            • 守护进程不能和用户进行交互
              • 没有输入和输出
            • 普通的进程运行 ./a.out
              • 程序在运行期间终端是阻塞的
            • 守护进程 ./a.out运行
              • 程序不阻塞终端

          • 守护进程作用:

            • 在后台周期性的执行某些特定操作

          • 注销计算机, 守护进程是不会终止的, 继续在后台运行

          # 1.2 进程组

          多个进程的集合. 所有的进程都有资格创建一个进程组, 也可以加入到某个进程组

          如果某些进程处理的操作相同, 可以将他们划分到同一个进程组中, 方便管理.

          一个进程组中至少有一进程

          • 进程组长的指定?

            • 进程组中的第一个进程就是组长

          • 进程组ID如何指定?

            • 进程组的组ID == 进程组长的PID

          • 函数

            // 获取当前进程所属的进程组的组ID
pid_t getpgrp(void);

// 获取指定进程所属的组的组ID
pid_t getpgid(pid_t pid);
	参数:指定的进程pid
    返回值:pid进程所属的组的组ID
        
// 将某个进程转移到另一个进程中 / 创建新的进程组
int setpgid(pid_t pid, pid_t pgid);
	参数:
		- pid: 进程ID, 要操作的进程
		- pgid: 进程组ID
			- pgid这ID是存在的, pid对应的进程被移动到这个进程组中
				- pid != pgid
			- pgid这ID是不存在的, 这个进程组就被创建了
				- pid == pgid

          # 1.3 会话

          多个进程组的集合.

          进程变成会话(会长)之后, 会脱离原来的终端 == 守护进程

          • 只有普通进程才能变成会话(会长), 会长也是一个进程
          • 组长是没有资格变成会话的

          • 函数

            // 获取进程所属的会话ID
#include <unistd.h>
pid_t getsid(pid_t pid);   

// 进程调用这个函数就可以变成会话
pid_t setsid(void);

          • 创建会话注意事项:

            1. 找一个没有任何职务进程
	- 使用当前进程创建一个新的子进程
	- 杀死父进程
2. 使用子进程调用 setsid() 函数 
	- 子进程变成了会长
	- 子进程就脱离终端了
3. 让子进程变成会长
	- 某些操作系统下需要使用管理权限
	- ubuntu下不需要

            //编写简单的守护进程
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>

int main()
{
    // 1. 创建子进程
    pid_t pid = fork();
    if(pid > 0)
    {
        // 父进程
        exit(0);
    }
    // 2. 子进程 -> 会长
    setsid();
    while(1)
    {
        sleep(100);
    }
    return 0;
}

          # 1.4 创建守护进程的步骤

          /*
	1. 创建一个新的子进程 - fork();
	2. 杀死父进程 -> exit(0);
	3. 将新的子进程提升为会话 -> setsid();
	4. 修改进程的工作目录 (可选)
		- 如何看进程的工作目录?
			- 在哪个目录下执行进程, 工作目录就是哪个进程
		- 为什么要修改呢?
			- 如果进程的工作目录在一个U盘里边, U盘被拔掉了 ==> 工作目录就没有了 ==> 进程不能正常工作
		- 如何修改?
			- int chdir(const char *path);
	5. 修改掩码 (可选)
		- 什么是掩码?
			- 修改创建的新文件的权限, 新文件的权限 = 0777 - 掩码的值
		- 修改的目的?
			- 在进程中修改掩码, 创建新文件, 这个文件的权限会受掩码的影响
				- 设置掩码为 0111
				- 创建新文件: 肯定没有执行权限  
		- 如何设置掩码?
			mode_t umask(mode_t mask);
	6. 关闭/重定向文件描述符 (可选)
		- 指定的是那个文件描述符?
			- 每个进程文件描述符表中的0, 1, 2默认是打开的, 对应当前终端
			- 虽然进程脱离了终端, 但是还是绑定着
		- 如何关闭:
			close(0);
			close(1);
			close(2);
		- 重定向:
			- 在linux的/dev目录中有一个空文件: /dev/null (设备文件)
			int fd = open("/dev/null", O_RDWR);
			dup2(fd, 0);
			dup2(fd, 1);
			dup2(fd, 2);
	7. 添加子进程的业务逻辑 (在后台不停的做某件事)
*/

          # 1.5 练习

          写一个守护进程, 每隔2s获取一次系统时间, 将这个时间写入到磁盘文件.

          #include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <sys/time.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <signal.h>
#include <time.h>

// 信号的回调
void writefile(int num)
{
    // 获取系统时间, s
    time_t sec = time(NULL); 
    // 对系统时间格式化
    struct tm* loctime = localtime(&sec);
    char* strtime = asctime(loctime);
    // 写磁盘文件
    int fd = open("time.txt", O_WRONLY|O_CREAT|O_APPEND, 0777);
    write(fd, strtime, strlen(strtime)+1);
    close(fd);
}

int main()
{
    // 1. 创建子进程
    pid_t pid = fork();
    if(pid > 0)
    {
        // 2. 杀死父进程
        exit(0);
    }
    // 3. 子进程提升为会话
    // 子进程脱离终端
    setsid();

    // 4. 修改进程的工作目录
    chdir("/home/robin");
    // 5. 修改掩码
    umask(0111);
    // 6. 重定向文件描述符
    int fd = open("/dev/null", O_RDWR);
    dup2(fd, 0);
    dup2(fd, 1);
    dup2(fd, 2);

    // 7. 核心处理
    // 初始化定时器
    // 设置定时器
    struct itimerval val;
    val.it_value.tv_sec = 2;
    val.it_value.tv_usec = 0;
    val.it_interval.tv_sec = 2;
    val.it_interval.tv_usec = 0;

    // 注册信号捕捉
    struct sigaction act;
    act.sa_flags = 0;
    act.sa_handler = writefile;
    sigemptyset(&act.sa_mask);
    sigaction(SIGALRM, &act, NULL);
    
    // 定时时间达到, 发 sigalrm
    setitimer(ITIMER_REAL, &val, NULL);
    while(1)
    {
        sleep(100);
    }

    return 0;
}

          # 2. 线程

          线程是轻量级的进程（LWP：light weight process），在Linux环境下线程的本质仍是进程。

          操作系统会以进程为单位，分配系统资源，所以我们也说，进程是资源分配的最小单位。线程是操作系统调度执行的最小单位.

          安装线程 man page，命令：

          sudo apt-get install manpages-posix-dev

          • 线程和进程的关系

            • 这个图是一个进程的虚拟地址空间
            • 在当前进程中创建了线程

            1. 在进程中创建线程
              • 原来的进程 -> 线程 -> 主线程(父线程)
              • 新的线程 -> 子线程
            2. 做了创建线程的操作, 虚拟地址空间没有被复制, 多个线程共用同一个地址空间
              • 内核区
                • 不共享
                  • 寄存器状态不共享
                  • 线程的ID
                • 共享
                  • 文件描述符
              • 用户区
                • 共享: ->用户区所有, 除了栈
                  • 代码区
                  • 全局数据区
                  • 堆区
                  • ....
                • 不共享
                  • 栈区
            3. 进程最小的资源分配单位
              • 进程启动之后, 操作系统会给其分配资源
              • 每个进程争抢cpu时间
            4. 线程是最小的任务执行单位
              • 创建新的线程, 不占用额外的系统资源
              • 线程在处理的任务的时候也需要争抢cpu时间
            5. 进行上下文切换的时候, 线程的效率更高

          • 任务调度 -> 下图是单核 -> 并发

            创建的线程并不是越多越高效, 什么时候效率最高?

            • 创建的线程数 == CPU的核心数, 相当于并行运行

          • 创建线程

            $man 3 pthread_create

            线程控制的函数都是标准c的库函数，以为Windows平台上也可以使用

            #include <pthread.h>

// pthread_t: 线程ID的类型, 无符号长整形
int pthread_create(pthread_t *thread, const pthread_attr_t *attr,
                          void *(*start_routine) (void *), void *arg);
	参数:
		- thread: 传出, 通过这个参数得到了创建出的新的线程的ID
		- attr: 线程属性, 一般使用默认值, 该参数指定为NULL
		- start_routine: 函数指针, 指向的函数就是子线程的处理动作
		- arg: 对应第三参数函数指针的参数, 回调函数被调用,这个参数作为实参传递到start_routine中
	返回值: 函数的状态
		- 0: 成功了
		- 失败了: 返回错误号 >0
// Compile and link with -pthread.
gcc *.c -l pthread

            // 因为返回的是错误号, 因此不能使用perror直接打印错误信息 == 不能使用perror
// 使用strerror
#include <string.h>
// 参数是得到的错误号, 返回值是错误号对应的错误信息
char *strerror(int errnum);

            // 创建线程的代码
// pthread_create.c 
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <pthread.h>

// 子线程的回调函数 == 子线程的处理逻辑
void* child(void* arg)
{
    printf("当前子线程的线程ID: %ld\n", pthread_self());
    // 子线程执行完毕, 直接退出
    pthread_exit(NULL);
    
    return NULL
}

int main()
{
    // 1. 创建子线程
    pthread_t tid;
    int ret = pthread_create(&tid, NULL, child, NULL);
    if(ret != 0){
        printf("创建线程失败, 原因: %s\n", strerror(ret));
        return 0;
    }
    // 2. 以下代码是父线程的
    printf("当前fu线程的线程ID: %ld\n", pthread_self());
    for(int i=0; i<5; ++i)
    {
        printf(" current i: %d\n", i);
    }
    
	// 主线程退出
    pthread_exit(NULL);
    
    // 这行代码不会执行
    exit(0);
    return 0;
}

          • 线程退出

            /*
	有多个线程:
		- 1个主线程
		- 若干个子线程
	线程退出的时候:
		- 如果其中子线程先退出, 对主线程和其他子线程没有影响
		- 主线程先退出, 运行的子线程强制被退出
			- 这些线程共用虚拟地址空间, 主线程退出的时候会销毁虚拟地址空间
*/
#include <pthread.h>
// 让某一个线程退出, 退出的时候对其他线程没有任何影响
void pthread_exit(void *retval);
	- 参数: 
		- retval: 线程退出的时候可以携带一些信息, 在这个指针指向的内存中
					- 在线程回收的时候, 可以将这些信息读出来

          • 回收子线程资源

            // 在子线程退出的时候, 主线程回收子线程资源
#include <pthread.h>
// 这是一个阻塞函数, 当有子线程还活着的时候, 子线程死了解除阻塞, 回收其资源
void* pt;
pthread_join(tid, &pt);
int pthread_join(pthread_t thread, void **retval);
	参数:
		- thread: 要回收的线程的线程ID
		- retval: 指向一级指针的地址, 传出参数, 里边的数据
			- pthread_exit(void *retval), 参数指针指向的数据
			- 如果对这个值不感兴趣, 直接写NULL
	返回值: 函数的状态
		- 0: 成功了
		- 失败了: 返回错误号 >0

            思路一：

            // 创建线程的代码
// pthread_create.c 
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <pthread.h>

struct Info{
	char name[24];
    int age;
};
struct Info info;  //全局
// 子线程的回调函数 == 子线程的处理逻辑
void* child(void* arg)
{
    printf("当前子线程的线程ID: %ld\n", pthread_self());
   	strcpy(info.name,"李磊");
    info.age = 20;

    pthread_exit(&info);
    
    return NULL
}

int main()
{
    // 1. 创建子线程
    pthread_t tid;
    int ret = pthread_create(&tid, NULL, child, NULL);
    if(ret != 0){
        printf("创建线程失败, 原因: %s\n", strerror(ret));
        return 0;
    }
    // 2. 以下代码是父线程的
    printf("当前fu线程的线程ID: %ld\n", pthread_self());
    for(int i=0; i<5; ++i)
    {
        printf(" current i: %d\n", i);
    }
    
    //回收资源
    void* pt;
    ret = pthread_join(tid,&pt);
    if(ret != 0){
        printf("回收线程失败, 原因: %s\n", strerror(ret));
        return 0;
    }
    struct Info* ptr = (struct Info*)pt;
    printf("name:%s,age = %d\n",ptr->name,ptr->age);
    
	// 主线程退出
    pthread_exit(NULL);
    return 0;
}

          • 线程分离

            // 在子线程被创建出之后, 设置线程分离, 主线程就不需要负责子线程的资源回收
// 如果父进程调用pthread_join()回收,直接报错
#include <pthread.h>

          // 如果多线程操作共享资源, 数据需要在cpu和物理内存之间同步,如果寄存器中的数据不能被及时同步到物理内存, 就会出现数据混乱
/*
	线程A, 线程B
	线程A抢到了cpu, 开始数数, cpu中数的 0->1, 将新的数值写回到物理内存  (理想状态)
	线程A抢到了cpu, 开始数数, cpu中数的 1->5, 还没来得及写回到物理内存失去了cpu	
		- 值[5]在寄存器中
		
	线程B抢到了cpu, 开始数数, 将数从物理内存中读出, 在cpu中开始数
		- 物理内存中的数 == 1, 因此从1开始数
		- cpu中数的 1->8, 将新的数值写回到物理内存  (理想状态)
		- cpu中数的 8->10, 还没来得及写回到物理内存失去了cpu	
			- 值 [10] 在寄存器中
			
	线程A抢到了cpu, 恢复失去cpu之前的状态, 将寄存器中的5写回到物理内存
		- 物理内存变量的值: 8 -> 5
*/

    // 阻塞sigchld信号
    sigset_t myset;
    sigemptyset(&myset);
    sigaddset(&myset, SIGCHLD);
    // 将myset设置到内核的阻塞信号集
    sigprocmask(SIG_BLOCK, &myset, NULL);

    pid_t pid;
    // 2. 创建子进程
    for(int i=0; i<50; ++i)
    {
        pid = fork();
        if(pid == 0)
        {
            break;
        }
    }

    if(pid == 0)
    {

    }
    else if(pid > 0)
    {
         // 1. 注册信号捕捉
        struct sigaction act;
        act.sa_flags = 0;
        act.sa_handler = recyle;
        sigemptyset(&act.sa_mask);
        sigaction(SIGCHLD, &act, NULL);

        // 在父进程中解除对sigchld信号的阻塞
        sigprocmask(SIG_UNBLOCK, &myset, NULL);

       // 让父进程活着

    }
