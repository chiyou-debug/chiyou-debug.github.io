                # 线程同步

                # 1. 线程同步

                # 1.1 同步概念

                ​ 所谓同步，即同时起步，协调一致。不同的对象，对“同步”的理解方式略有不同。如，设备同步，是指在两个设备之间规定一个共同的时间参考；数据库同步，是指让两个或多个数据库内容保持一致，或者按需要部分保持一致；文件同步，是指让两个或多个文件夹里的文件保持一致。等等

                ​ 而，编程中、通信中所说的同步与生活中大家印象中的同步概念略有差异。“同”字应是指协同、协助、互相配合。主旨在协同步调，按预定的先后次序运行。

                # 1.2 线程同步

                • 线程同步，指一个线程发出某一功能调用时，在没有得到结果之前，该调用不返回。同时其它线程为保证数据一致性，不能调用该功能。

                举例1：银行存款 5000。柜台，折：取3000；提款机，卡：取 3000。剩余：2000

                ![1564644834918](./images/Synchronization mechanism.png>)

                举例2： 内存中100字节，线程T1欲填入全1， 线程T2欲填入全0。但如果T1执行了50个字节失去cpu，T2执行，会将T1写过的内容覆盖。当T1再次获得cpu继续 从失去cpu的位置向后写入1，当执行结束，内存中的100字节，既不是全1，也不是全0。

                • 产生的现象叫做“与时间有关的错误”(time related)。为了避免这种数据混乱，线程需要同步。

                • “同步”的目的，是为了避免数据混乱，解决与时间有关的错误。实际上，不仅线程间需要同步，进程间、信号间等等都需要同步机制。

                • 因此，所有“多个控制流，共同操作一个共享资源”的情况，都需要同步。

                • 临界资源：一次只允许一个线程使用的资源。

                  • 线程共享的资源都称之为临界资源
                  • 在程序中就是全局变量, 堆变量 ...

                • 临界区

                  一个访问共用资源的程序片段，而这些共用资源又无法同时被多个线程访问，这个区域称之为临界区。当有线程进入临界区段时，其他线程或必须等待，以确保这些共用资源是被互斥使用的。

                # 1.3 数据混乱原因

                1. 资源共享（独享资源则不会）
                2. 调度随机（意味着数据访问会出现竞争）
                3. 线程间缺乏必要的同步机制。

                ​ 以上3点中，前两点不能改变，欲提高效率，传递数据，资源必须共享。只要共享资源，就一定会出现竞争。只要存在竞争关系，数据就很容易出现混乱。

                ​ 所以只能从第三点着手解决。使多个线程在访问共享资源的时候，出现互斥。

                # 2. 互斥锁(量)

                • 互斥锁类型?

                  pthread_mutex_t //其本质是一个结构体。为简化理解,应用时可忽略其实现细节,简单当成整数看待。

                • 借助mutex(互斥量、互斥锁)管理共享数据实现同步的一般步骤:

                  pthread_mutex_t mutex;		//定义一把互斥锁 
pthread_mutex_init();       //初始化锁   	   1
pthread_mutex_lock();      	//加锁			1----> 0
//访问共享数据
pthrad_mutext_unlock();     //解锁	     	0++	--> 1
pthead_mutex_destroy();     //销毁锁

//注意事项:
	尽量保证锁的粒度,越小越好。(访问共享数据前,加锁。访问结束【立即】解锁)

                • 互斥锁相关函数

                  restrict: 关键字, 修饰一个指针(地址)
    - mutex指针指向的地址只能由mutex这个指针进行操作, 其他指针不行
    - 举例:
		pthread_mutex_t *ptr = mutex;	// error
		通过ptr操作mutex指针指向的内存
            
#include <pthread.h>
// 初始化互斥锁, 初始化完毕之后, 互斥锁的打开的
// 动态初始化:
int pthread_mutex_init(pthread_mutex_t *restrict mutex,
           const pthread_mutexattr_t *restrict attr);
	参数:
		- mutex: 要初始化的互斥锁的地址
		- attr: 出斥锁的属性, 一般使用默认属性不设置, 写NULL
//静态初始化:
pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;

// 销毁互斥锁		
int pthread_mutex_destroy(pthread_mutex_t *mutex);

// 1. 如果互斥锁状态是打开的, 上锁成功, 锁被锁定
// 2. 如果互斥锁状态已经被锁定了, 线程被阻塞在这把锁上, 当这把锁被打开之后, 线程解除阻塞
int pthread_mutex_lock(pthread_mutex_t *mutex);

// 1. 如果互斥锁状态是打开的, 上锁成功, 锁被锁定
// 2. 如果互斥锁状态已经被锁定了,线程不会阻塞在这把锁上, 这个函数不阻塞线程
// 3. 通过返回值判断加锁成功了还是失败了
// 		0-> 成功, 非0 -> 失败直接返回错误号
int pthread_mutex_trylock(pthread_mutex_t *mutex);

// 打开互斥锁
int pthread_mutex_unlock(pthread_mutex_t *mutex);

                • 互斥锁特点?

                  ## 线程对临界区的访问是串行的, 访问效率低
	- 读
	- 写
## 锁本身只是"建议锁",对公共数据进行保护。所有线程【应该】在访问公共数据前先拿锁再访问。但，锁本身不具备强制性。

                  示例

                  #include <stdio.h>
#include <string.h>
#include <pthread.h>
#include <stdlib.h>
#include <unistd.h>

//在全局区定义一把互斥锁
pthread_mutex_t mutex;

void *tfn(void *arg)
{
    srand(time(NULL));

    while (1) {
        pthread_mutex_lock(&mutex);   //加锁
        printf("hello ");
        sleep(rand() % 3);	/*模拟长时间操作共享资源，导致cpu易主，产生与时间有关的错误*/
        printf("world\n");
        sleep(rand() % 3);
        pthread_mutex_unlock(&mutex); //解锁
    }

    return NULL;
}

int main(void)
{
    pthread_t tid;
    srand(time(NULL));
	//初始化锁
    pthread_mutex_init(&mutex, NULL);
	//创建子线程
    pthread_create(&tid, NULL, tfn, NULL);
    while (1) {
        pthread_mutex_lock(&mutex);	//加锁
        printf("HELLO ");
        sleep(rand() % 3);
        printf("WORLD\n");
        sleep(rand() % 3);
        pthread_mutex_unlock(&mutex);//解锁
    }
    
    //回收子线程资源
    pthread_join(tid, NULL);
    //销毁锁
    pthread_mutex_destroy(&mutex);

    return 0;
}
/*线程之间共享资源stdout*/

                # 3. 死锁

                两个或两个以上的进程在执行过程中，因争夺共享资源而造成的一种互相等待的现象，若无外力作用，它们都将无法推进下去。此时称系统处于死锁状态或系统产生了死锁 。

                • 死锁 的几种场景

                  • 忘记释放锁，自己将自己锁住

                    int func()
{
    while(1)
    {
        // 再次进出入的时候, 会阻塞
        pthread_mutex_lock(&mutex);
        xxxxxx;
        xxxxx;
        // 忘记解锁
        // pthread_mutex_unlock();
    }
}

                  • 单线程重复申请锁

                    int funcA()
{
   	pthread_mutex_lock(&mutex);
	xxxx;
    xxxxx;
    pthread_mutex_unlock(&mutex);
}

int func()
{
    while(1)
    {
        // 再次进出入的时候, 会阻塞
        pthread_mutex_lock(&mutex);
        // 对一个mutex锁两次
        funcA();
        xxxxxx;
        xxxxx;
        pthread_mutex_unlock(&mutex);
    }
}

                  • 多线程多锁申请， 抢占锁资源

                    线程A有一个锁1，线程B有一个锁2。线程A试图调用lock来获取锁2就得挂起等待线程B释放，线程B也调用lock试图获得锁1。那么这就很尴尬，都在等对方释放，然后获得对方的锁。

                    共享资源有两个:

                    • x -> 锁A
                    • y -> 锁B

                    情况:

                    • 线程A使用A锁对资源X加锁成功了
                    • 线程B使用B锁对资源Y加锁成功了
                    • 这两个线程都没有解锁
                      • 线程A访问资源Y, 被B锁阻塞了
                      • 线程B访问资源X, 被A锁阻塞了
                    • 这两个线程都阻塞了, 并且无法解除阻塞

                  • 死锁的解决

                    • 多检查代码

                    • 在加锁的时候使用trylock

                    • 对其他资源加锁的时候, 先将自己拥有的锁解开

                # 4. 读写锁

                • 读写锁类型? 是几把锁?

                  // 类型:
pthread_rwlock_t
// 是几把锁?
是一把锁, 有两个功能
	- 专门锁定读操作
	- 专门锁定写操作

                • 读写锁的特点?

                  对读写锁的操作 -> 读共享, 写独占, 如果读写都有, 写的优先级高
	- 多个位置加了读锁, 这些操作是并行的
	- 如果多个位置加了写锁, 写操作是串行的
	- 如果既加了读锁, 也加了写锁, 加写锁的线程会优先加锁成功
// 场景
    // 1. 线程A加读锁成功, 又来了三个线程, 做读操作, 可以加锁成功
    	- 都是读操作, 读锁是共享的
    // 2. 线程A加写锁成功, 又来了三个线程, 做读操作, 三个线程阻塞
    	- 写锁是独占的, 三个读线程要阻塞
    	- 写锁被解锁了, 三个线程的读是并行的, 读锁是共享的
    //	  线程A加读锁成功, 又来了三个线程, 做写操作, 三个线程阻塞
    	- 读写不能同时进行, 写线程来的晚没有抢到锁, 写线程阻塞
    	- 读线程将读锁解锁, 三个写的线程优先级相同, 抢锁, 写是串行的
    // 3. 线程A加读锁成功, 又来了B线程加写锁阻塞, 又来了C线程加读锁阻塞
    	- B线程和C线程阻塞
    	- 线程A解锁, 线程B加锁成功 -> 写, 线程C阻塞
    	- 线程B解锁, 线程C加锁成功 -> 读, 线程C解锁

                • 什么时候使用读写锁?

                  程序中有大量的对共享资源进行读操作的情况

                • 操作函数

                  #include <pthread.h>
// 初始化读写锁
// 动态初始化
int pthread_rwlock_init(pthread_rwlock_t *restrict rwlock,
           const pthread_rwlockattr_t *restrict attr);
	参数:
		- rwlock: 要初始化的读写锁
		- attr: 读写锁的属性, 默认属性指定=NULL
// 静态初始化:
pthread_rwlock_t rwlock = PTHREAD_RWLOCK_INITIALIZER;
            
// 释放读写锁资源
int pthread_rwlock_destroy(pthread_rwlock_t *rwlock);
// 加读锁, 这是一个阻塞函数
// 什么情况下阻塞?
// 	- 当有线程通过这把锁锁定了写, 这时候锁定读就阻塞
//  - 当有线程通过这把锁锁定了读, 再次加读锁 -> 不阻塞
int pthread_rwlock_rdlock(pthread_rwlock_t *rwlock);
// 尝试加读锁, 如果被写锁定了, 尝试加读锁失败 -> 不阻塞, 直接返回了
int pthread_rwlock_tryrdlock(pthread_rwlock_t *rwlock);
// 加写锁, 这是一个阻塞函数
// 什么情况下阻塞?
// 	- 当有线程通过这把锁锁定了写, 这时候锁定写就阻塞
//  - 有线程通过这把锁锁定了读, 再次加写锁 ->阻塞
//  - 如果这把锁是打开的, 加锁成功 -> 不阻塞
int pthread_rwlock_wrlock(pthread_rwlock_t *rwlock);
// 尝试加写锁, 加锁失败直接返回 -> 不阻塞
int pthread_rwlock_trywrlock(pthread_rwlock_t *rwlock);
// 将读写锁解锁
int pthread_rwlock_unlock(pthread_rwlock_t *rwlock);

以上所有的返回值:
	成功: 0
    失败: != 0, 错误号
通过函数: char* strerror(errno);

                • 练习

                  ## 8个线程操作同一个全局变量
## 3个线程不定时写同一全局资源，5个线程不定时读同一全局资源
	- 写不使用线程同步
	- 线程同步

                  线程不同步

                  #include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <pthread.h>

int number = 0;
// 写数据
void* writeNum(void* arg)
{
    while(1)
    {
        number++;
        printf("+++write,tid:%ld,numver:%d\n",pthread_self(),number);
        usleep(rand()%100);
    }
    return NULL;
}

// 读数据
void* readNum(void* arg)
{
    while(1)
    {
        printf("---read,tid:%ld,number:%d\n",pthread_self(),number);
        usleep(rand()%100);
    }
}

int main()
{
    // 创建线程
    // 3写
    pthread_t wtid[3];
    for(int i = 0;i<3;i++)
    {
        pthread_create(&wtid[i],NULL,writeNum,NULL);
    }

    // 5读
    pthread_t rtid[5];
    for(int i = 0;i<5;i++)
    {
        pthread_create(&rtid[i],NULL,readNum,NULL);
    }

    for(int i = 0; i<3;i++)
    {
        pthread_join(wtid[i],NULL);
    }

    for(int i = 0; i<5;i++)
    {
        pthread_join(rtid[i],NULL);
    }

    return 0;
}

                  线程同步

                  #include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <pthread.h>

int number = 0;
// 读写锁变量
pthread_rwlock_t rwlock;
// 写数据
void* writeNum(void* arg)
{
    while(1)
    {
        // 写操作, 加写锁
        pthread_rwlock_wrlock(&rwlock);
        number++;
        printf("+++write, tid: %ld, number: %d\n", pthread_self(), number);
        pthread_rwlock_unlock(&rwlock);
        usleep(rand()%100);
    }
    return NULL;
}

// 读数据
void* readNum(void* arg)
{
    while(1)
    {
        pthread_rwlock_rdlock(&rwlock);   // 读线程间,读锁共享
        printf("---read, tid: %ld, number: %d\n", pthread_self(), number);
        pthread_rwlock_unlock(&rwlock);
        usleep(rand()%100);
    }
    return NULL;
}

int main()
{
    // 初始化读写锁
    pthread_rwlock_init(&rwlock, NULL);
    // 创建线程
    // 3写
    pthread_t wtid[3];
    for(int i=0; i<3; ++i)
    {
        pthread_create(&wtid[i], NULL, writeNum, NULL);
    }
    // 5读
    pthread_t rtid[5];
    for(int i=0; i<5; ++i)
    {
        pthread_create(&rtid[i], NULL, readNum, NULL);
    }

    for(int i=0; i<3; ++i)
    {
        pthread_join(wtid[i], NULL);
    }
    for(int i=0; i<5; ++i)
    {
        pthread_join(rtid[i], NULL);
    }

    // 销毁读写锁
    pthread_rwlock_destroy(&rwlock);

    return 0;
}

                # 5. 条件变量 - condition

                • 条件变量是锁吗?

                  不是锁, 但是需要和互斥锁一块使用
	- 条件变量可以阻塞线程
	- 互斥锁主要做数据同步

                • 条件变量两个动作

                  // 当条件不满足的时候, 线程不能工作, 让线程阻塞
// 当条件再次满足的时候, 让线程解除阻塞, 恢复工作

                  关于pthread_cond_wait的解析

                  阻塞等待一个条件变量：

                  ​ int pthread_cond_wait(pthread_cond_t *restrict cond，pthread_mutex_t *restrict mutex);

                  函数作用：

                  • 阻塞等待条件变量cond(参1)满足
                  • 释放已掌握的互斥锁(解除互斥量)，相当于pthread_mutex_unlock(&mutex);
                    • 1.2两步为一个原子操作
                  • 当被唤醒，pthread_cond_wait函数返回时，解除阻塞并重新申请获取互斥锁
                  void* customer(void* arg)
  {
      while(1)
      {   
          pthread_mutex_lock(&mutex);
          // 1. 从头部取出一个节点, 打印信息
          struct Node* pcur = head;
          if(pcur == NULL)
          {   
              // 链表中没有节点了
              // 阻塞消费者线程 
              // 1. 如果pthread_cond_wait阻塞了线程, 会将线程加锁成功的互斥锁解锁 
              // 2. 解锁成功之后, 阻塞线程
              // 3. 当线程从该函数上解除阻塞的时候, 这个线程会再次对互斥锁加锁
              pthread_cond_wait(&cond, &mutex);
          }   
          printf("node number: %d\n", pcur->data);
          // 2. head指针后移
          head = head->next;
          // 3. 将这个节点从链表中删除
          free(pcur);
          pthread_mutex_unlock(&mutex); 
  
          sleep(rand()%3);
      }   
  }

                # 6. 信号量 - semaphore

                信号量用在多线程多任务同步的，一个线程完成了某一个动作就通过信号量告诉别的线程，别的线程再进行某些动作。

                信号量不一定是锁定某一个资源，而是流程上的概念，比如:有A,B两个线程，B线程要等A线程完成某一任务以后再进行自己下面的步骤，这个任务并不一定是锁定某一资源，还可以是进行一些计算或者数据处理之类。

                信号量(信号灯)与互斥锁和条件变量的主要不同在于"灯"的概念,灯亮意味着资源可用,灯灭则意味着不可用。

                信号量主要阻塞线程, 不能完全保证线程安全.

                如果要保证线程安全, 需要信号量和互斥锁一起使用.

                • 信号量类型

                  #include <semaphore.h>
sem_t

- 生产者对应一个信号量: semp
	- 通过这个变量中的值控制线程是阻塞还是不阻塞
- 消费者对应一个信号量: semc

                • 信号量操作函数

	参数:
		- sem: 被初始化的信号量变量, 这个参数看做一个整形数即可
		- pshared: 
			- 0: 线程
			- 1: 进程
		- value: 给第一个参数sem赋值, 通过这个变量中的值控制线程是阻塞还是不阻塞

                • 练习

                  1. 生产者和消费者模型需要两个信号量
	- 生产者: psem
  		// 意味着生产者刚开始的时候是不阻塞的
		// 可以有5个生产者同时投入到生产状态
  		sem_init(&psem, 0, 5);
	- 消费者: csem
		// 意味着消费者刚开始的时候是阻塞的, 因为没有东西
		sem_init(&csem, 0, 0);
2. 生产者开始生产产品
	- 其中某一个线程开始工作
		// psem中的值最终 == 0, 所有的生产者都被阻塞了
		sem_wait(&psem);	
	- 创建一个新节点, 添加到表头
	- 通知消费者消费
		sem_post(&csem);
3. 消费者消费
	- 某一个消费者线程去消费
		sem_wait(&csem);
	- 消费完成之后, 产品没有了, 通知生产者生产
		sem_post(&psem);

                  示例:

                  #include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <pthread.h>
#include <semaphore.h>

// 链表节点的结构体
struct Node
{
    int data;
    struct Node* next;
};
// 创建第一个节点就是数据节点的链表
// 定义一个指针, 永远指向头结点
struct Node* head = NULL;

// 创建互斥锁变量
pthread_mutex_t mutex;
// 信号量
sem_t psem;
sem_t csem;

// 生产者回调
void* product(void* arg)
{
    while(1)
    {
        // 生产者生产,消耗一个资源
        sem_wait(&psem);
        pthread_mutex_lock(&mutex);
        // 1. 创建新节点
        struct Node* pNew = (struct Node*)malloc(sizeof(struct Node));
        // 2. 新节点初始化
        pNew->data = rand() % 1000;
        // 3. 新节点挂到链表的头部
        pNew->next = head;
        head = pNew;
        printf("++++++ tid: %ld, number: %d\n", pthread_self(), pNew->data); 
        pthread_mutex_unlock(&mutex);

        // 通知消费者消费
        sem_post(&csem);

        sleep(rand()%3);
    }

    return NULL;
}

// 消费者回调
void* customer(void* arg)
{
    while(1)
    {
        // 消费者消费, 消耗一个资源
        sem_wait(&csem);
        pthread_mutex_lock(&mutex);
        struct Node* pcur = head;
        printf("----- tid: %ld, number: %d\n", pthread_self(), pcur->data); 
        // 2. head指针后移
        head = head->next;
        // 3. 将这个节点从链表中删除
        free(pcur);
        pthread_mutex_unlock(&mutex);

        // 通知生产者生产
        sem_post(&psem);

        sleep(rand()%3);
    }
}

int main()
{
    // 初始化生产者信号量
    sem_init(&psem, 0, 5);
    // 初始化消费者信号量
    sem_init(&csem, 0, 0);
    pthread_mutex_init(&mutex, NULL);
    // 创建生产者线程
    pthread_t tid[5];
    // 创建消费者线程
    pthread_t ctid[5];
    for(int i=0; i<5; ++i)
    {
        pthread_create(&tid[i], NULL, product, NULL);
        pthread_create(&ctid[i], NULL, customer, NULL);
    }
	// 回收线程资源
    for(int i=0; i<5; ++i)
    {
        pthread_join(tid[i], NULL);
        pthread_join(ctid[i], NULL);
    }
    // 销毁信号量与互斥锁
    sem_destroy(&psem);
    sem_destroy(&csem);
    pthread_mutex_destroy(&mutex);

    return 0;
}
