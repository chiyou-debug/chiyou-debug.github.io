    # 1. 内存四区图PlusPlus

    # 1.1 虚拟地址空间

    假设有一个磁盘可执行程序 app, ./app执行这个程序, 得到了一个进程

    • 进程: 不占用磁盘空间, 占用内存, cpu资源
    • 进程启动之后, 程序中的数据不会被全部加载到内存中
      • cpu在处理进程的时候, 使用什么数据, 将什么数据加载到内存
        • 减少了内存消耗
      • 在程序中打印的内存地址, 并不是物理内存的地址, 是虚拟地址空间中的地址
        • 数据在物理内存中的地址对进程不可见
          • 严重威胁数据的安全性

    • 虚拟地址空间的大小

      // 如果是32位操作系统: 
	2 的 32 次方 ≈ 4G
// 如果是64位操作系统: 
	2 的 64 次方 ≈ 特别大的数G
// 以上的值都是理论值, 实际占用的大小要小的多

    • 虚拟地址空间在什么地方?

      • 虚拟地址空间 == 虚拟内存
      • 在磁盘上

    • 虚拟地址空间中的分区

      • 内核区

        ## 在一个操作系统中, 内核只有一个
## 在一个系统中, 运行着多个进程, 它们共用同一个内核区
## 占用虚拟地址空间中最上边 1个G 的大小
## 特点: 作为使用者不能直接对内核区进行读写操作,否者直接出段错误

      • 用户区

        - 0-4k大小: 受保护的地址 NULL
- .text: 代码段
- .data: 初始化的全局变量
- .bss:  未初始化的全局变量
- 堆: 动态申请内存(malloc, new)
- 动态库加载区: 加载动态库
- 栈: 局部变量
- 命令行参数:
	int main(int argc, char* argv[]);
- 环境变量:
	每个进程都有属于自己的环境变量

    # 1.2 文件描述符表

    Linux中一切皆文件, 因此在linxu中会进行大量的文件操作

    • 如何判断这个文件被程序(进程) 打开了?
      • 通过进程控制块中的文件描述表来记录的
        • 每打开一磁盘文件就会占用文件描述符表中的一个文件描述符
          • 文件描述符表 == 整形数组
          • 一个文件描述符 == 数组中的一个元素, 就是一个整形数
            • 通过文件描述符就可以对打开的磁盘文件进行编号
              • 通过这个编号就可以间接找到磁盘文件

    ## 每个进程中都有一文件描述符表, 默认可以打开多少个文件?
	- 1024个, 默认值, 可以调整
## 默认文件描述符是不是空的?
	- 不是, 当一个进程启动之后, 默认和终端是有关联的, 绑定在一起了
		- 程序可以接收终端的输入
		- 通过终端进行输出
		- 为什么终端是文件?
			- linux中一切皆文件, 终端设备会被抽象成文件放在 `/dev`
			- 当前终端的设备名叫: `/dev/tty`
	- 和终端绑定的时候占用了三个文件描述符
		- 标准输入: stdin  -> 0  STDIN_FILENO
		- 标准输出: stdout -> 1  STDOUT_FILENO
		- 标准错误: stderr -> 2  STDERR_FILENO
## 如果打开一个磁盘文件, 从文件描述符表中找一个最小的没有被占用的位置标记被打开的文件

      ## 1. 使用标准C库函数打开磁盘文件

	- 文件描述符

	- 标准C库 I/O 函数 都有缓冲区, 默认是 8k
	- linux系统函数没有缓冲区
	- 在linux中如果调用标准C库函数, linux系统函数会在底层被调用

    • 标准C 库IO和 Linux系统IO的关系

      // 读操作
/*
	- 通过fputs, 写了一个字符串, 数据进入函数对应的缓冲区中
	- 当该函数对应的缓冲区被写满了, 数据被linux系统函数写入到磁盘
	- 当该函数对应的缓冲区没有被写满, 但是做了额外的强制刷新的操作, 数据被linux系统函数写入到磁盘
	- 假如fputs被调用了100次, write被调用次数会小于100
*/

// 写操作
/*
	- 通过fgets去磁盘读数据
	- fgets调用了系统函数read, read将读出的数据放入fgets的缓冲区
	- fgets再次调用的时候取的数据在缓冲区中
	- fgets的缓冲区中有数据read不工作, 没数据了read读磁盘
*/

//注意：
// 有缓冲区
printf("hello,world"); //因为有缓冲区,所以有时候打印的时候终端不显示
// 行缓冲, 使用\n可以将数据强制从缓冲区中刷出
printf("hello,world\n");

// open函数只有一个, 是变参函数
// 作用: 打开或者创建一个文件
// 打开文件
int open(const char *pathname, int flags);
	参数:
		- pathname: 要打开的文件的路径(相对/绝对)
        - flag: 打开文件的方式
        	- 三选一(必须):
                - O_RDONLY: 只读
                - O_WRONLY: 只写
                - O_RDWR: 读写
            - 附加属性:
        		- O_APPEND: 写文件的时候追加
        		- O_CREAT: 创建新文件
                - O_EXCL: 检测文件是否存在
                 	- 必须和O_CREAT一起使用
                 		- O_EXCL|O_CREAT
                 	- 如果文件存在, open调用失败, 返回-1,设置errno
                    - 如果文件不存在, 创建新文件
     返回值:
		- 成功: 文件描述符
		- 失败: -1,设置errno
// 创建文件
int open(const char *pathname, int flags, mode_t mode);
	参数3: 前提是参2值定了O_CREAT
		- mode: 传八进制的整形数, 指定新文件的权限
			- 文件所有者/所属组/其他人对文件的访问权限
			- 最终权限需要计算: (mode & ~umask) //umask 文件权限屏蔽码 022
				777				111111111
                775				111111101

      #include <unistd.h>

	参数:
		- fd: open函数的返回值, 也就是要写的文件对应的文件描述符
		- buf: 存储了要写入磁盘的数据

      /* ./mycp src dst 命令行参数实现简单的cp命令 */
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <fcntl.h>

int main(int argc,char* argv[])
{
    int fd1 = open(argv[1],O_RDONLY);
    int fd2 = open(argv[2],O_RDWR|O_CREAT|O_TRUNC,0664);
    if(fd1 == -1 || fd2 == -1){
        perror("open argv error");
        exit(1);
    }

    int len = 0;
    char buf[1024];
    while((len = read(fd1,buf,sizeof(buf)))!=0)
    {
        if(len < 0){
            perror("read error");
            break;
        }
        write(fd2,buf,len);
    }

    close(fd1);
    close(fd2);

    return 0;
}

    • lseek

      int fseek(FILE *stream, long offset, int whence);

#include <sys/types.h>
#include <unistd.h>
// 移动文件指针
off_t lseek(int fd, off_t offset, int whence);
	参数:
		- fd: open函数的返回值, 也就是要操作文件对应的文件描述符
		- offset: 偏移量, 整形数, 代表的意思要结合第三个参数看
		- whence: 
			- SEEK_SET: 设置文件指针的位置
			- SEEK_CUR: 当前文件指针的位置 + 第二个参数的偏移量
			- SEEK_END: 文件的大小 + 第二个参数的偏移量
	返回值:
		- 成功: 文件指针最后的位置
		- 失败: -1
// 三个操作
// 1. 移动到文件头
lseek(fd, 0, SEEK_SET);         
// 2. 获取文件指针当前的位置
lseek(fd, 0, SEEK_CUR); 
// 3. 获取文件长度  
lseek(fd, 0, SEEK_END); 
// 4. 拓展文件大小
lseek(fd, 100, SEEK_END); 
// 如果想要文件拓展成功, 必须要再进行一次写操作
write(fd, "a", 1);

      示例

      /*使用lseek进行文件拓展*/
#include <stdio.h>
#include <unistd.h>
#include <fcntl.h>

int main()
{
    // 打开文件
    int ret = open("abc.txt",O_RDWR|O_CREAT,0664);
    if(ret == -1)
    {
        perror("open");
    }

    // 进行文件拓展
    lseek(ret,1000,SEEK_END);
    write(ret,"end",3);

    close(ret);

    return 0;
}

    # 3.1 stat/lstat函数

    ## 创建快捷方式
ln -s 文件的绝对路径  快捷方式的名字

    // 函数原型 -> 查看文件的属性信息
#include <sys/types.h>
#include <sys/stat.h>
#include <unistd.h>

int stat(const char *pathname, struct stat *buf);
	参数:
		- pathname: 要操作的文件的路径和名字 /home/robin/a.txt   ./a.txt
		- buf: 结构体指针, 传出参数, stat函数会将得到的属性信息写入该指针对应的内存中

struct stat {
    dev_t          st_dev;      //文件的设备编号
    ino_t          st_ino;      //inode节点
    mode_t         st_mode;     //文件的类型和存取的权限, 16位整形数
    nlink_t        st_nlink;    //连到该文件的硬连接数目，刚建立的文件值为1
    uid_t          st_uid;      //用户ID
    gid_t          st_gid;      //组ID
    dev_t          st_rdev;     //(设备类型)若此文件为设备文件，则为其设备编号
    off_t          st_size;     //文件字节数(文件大小)
    blksize_t      st_blksize;  //块大小(文件系统的I/O 缓冲区大小)
    blkcnt_t       st_blocks;   //块数
    time_t         st_atime;    //最后一次访问时间
    time_t         st_mtime;    //最后一次修改时间(文件内容)
    time_t         st_ctime;    //最后一次改变时间(指属性)
};

    • 获取文件大小

      #include <sys/stat.h>
void test()
{
    struct stat st;

    stat("abc.txt", &st);
    printf("abc.txt size: %d\n", (int)st.st_size);
}

    • 获取文件类型

      //内建宏：
    S_ISREG(m)  is it a regular file?
    S_ISDIR(m)  directory?
    S_ISCHR(m)  character device?
    S_ISBLK(m)  block device?
    S_ISFIFO(m) FIFO (named pipe)?
    S_ISLNK(m)  symbolic link?  (Not in POSIX.1-1996.)
    S_ISSOCK(m) socket?  (Not in POSIX.1-1996.)
    
void test()
{
    struct stat st;

    stat("abc.txt", &st);
    // 判断文件类型
    if(S_ISREG(st.st_mode))
    {
        printf("该文件是一个普通文件\n");
    }
    else if(S_ISDIR(st.st_mode))
    {
        printf("这是一个目录\n");
    }
    printf("abc.txt size: %d\n", (int)st.st_size);
}

    • 获取对文件的操作权限

      // 判断文件所有者对文件有没有执行权限
// 判断其他人对文件有没有写权限
if(S_IWOTH & st.mode)
{
 	// 有这个权限 
}
else
{
    // 没有这个权限
}
关于变量 st_mode: 
	- st_mode -- 16位整数
		○ 0-2 bit -- 其他人权限
			- S_IROTH    00004  读权限
			- S_IWOTH    00002  写权限
			- S_IXOTH    00001  执行权限
			- S_IRWXO    00007  掩码, 过滤 st_mode中除其他人权限以外的信息
		○ 3-5 bit -- 所属组权限
			- S_IRGRP    00040  读权限
			- S_IWGRP    00020  写权限
			- S_IXGRP    00010  执行权限
			- S_IRWXG    00070  掩码, 过滤 st_mode中除所属组权限以外的信息
		○ 6-8 bit -- 文件所有者权限
			- S_IRUSR    00400    读权限
			- S_IWUSR    00200    写权限
			- S_IXUSR    00100    执行权限
			- S_IRWXU    00700    掩码, 过滤 st_mode中除文件所有者权限以外的信息
		○ 9-11 bit --  特殊权限位(了解)
		○ 12-15 bit -- 文件类型
			- S_IFSOCK   0140000 套接字
			- S_IFLNK    0120000 符号链接（软链接）
			- S_IFREG    0100000 普通文件
			- S_IFBLK    0060000 块设备
			- S_IFDIR    0040000 目录
			- S_IFCHR    0020000 字符设备
			- S_IFIFO    0010000 管道
			- S_IFMT     0170000 掩码,过滤 st_mode中除文件类型以外的信息
		(st_mode & S_IFMT) ==  S_IFREG

    • 符号穿透：stat会，lstat不会。

	参数是 opendir 的返回值

      示例：读指定目录下普通文件的个数

      #include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <dirent.h>

int getRegFiles(const char* path)
{
    // 1. 打开目录
    DIR* dir = opendir(path);
    if(dir == NULL)
    {
        perror("opendir");
        return 0;
    }
    // 2. 遍历当前目录
    struct dirent* ptr = NULL;
    int count = 0;
    while((ptr = readdir(dir)) != NULL)
    {
        // 如果是目录 . ..
        if(strcmp(ptr->d_name, ".") ==0 ||
           strcmp(ptr->d_name, "..") == 0)
        {
            continue;
        }
        // 假设读到的当前文件是目录
        if(ptr->d_type == DT_DIR)
        {
            // 目录
            char newPath[1024];
            sprintf(newPath, "%s/%s", path, ptr->d_name);
            // 读当前目录的子目录
            count += getRegFiles(newPath);
        }
        else if(ptr->d_type == DT_REG)
        {
            // 普通文件
            count ++;
        }
    }

    closedir(dir);
    return count;
}

int main(int argc, char* argv[])
{
    // ./a.out path
    if(argc < 2)
    {
        printf("./a.out path\n");
        return 0;
    }

    int num = getRegFiles(argv[1]);
    printf("%s 目录中普通文件个数: %d\n", argv[1], num);

    return 0;
}

    # 3.3 dup、dup2、fcntl

    这三个函数都能复制文件描述符

    • 一个文件描述符对应一个打开的磁盘文件 : fd1 -> a.txt
    • 复制了文件描述符fd1, 得到了fd2
    • fd2也可以操作磁盘文件a.txt
    • fd1和fd2是两个独立的文件描述符

    • dup

      #include <unistd.h>
// 复制文件描述符
// 返回值和参数指向的是同一个磁盘文件
int dup(int oldfd);
	参数: 要被复制的文件描述符
 	返回值: 被复制出的文件描述符

      示例

      #include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <fcntl.h>

int main()
{
    int ret = open("abc.txt", O_RDWR|O_CREAT, 0777);
    if(ret == -1)
    {
        perror("open");
        return 0;
    }

    // 复制ret
    int fd = dup(ret);
    close(ret);

    printf("open return value ret: %d\n", ret);
    printf("open return value fd: %d\n", fd);

    // 写文件
    char* pt = "文件描述符复制, 测试.....";
    write(fd, pt, strlen(pt));

    close(fd);

    return 0;
}

    • dup2

      #include <unistd.h>
/*
	场景1:
		oldfd -> a.txt
		newfd -> b.txt
		函数调用成功:
			newfd不再指向b.txt, newfd指向了a.txt
	场景2:
		oldfd -> a.txt
		newfd -> 不指向任何文件
		函数调用成功:
			newfd指向了a.txt
	场景3:
		oldfd -> a.txt
		newfd -> a.txt
		函数调用成功:
			什么事儿也不干
*/
// 复制文件描述符/重定向文件描述符
int dup2(int oldfd, int newfd);
	参数oldfd: 要被拷贝的文件描述符
	参数newfd: 要将oldfd中的信息拷贝到newfd中
	返回值返回 newfd 的值

      示例

      #include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <fcntl.h>

int main()
{
    int ret = open("abc.txt", O_RDWR|O_CREAT, 0777);
    if(ret == -1){
        perror("open");
    }

    int fd2 = open("test.txt", O_RDWR|O_CREAT, 0777);
    if(fd2 == -1){
        perror("open");
    }

    // 复制ret
    dup2(ret, fd2);
    close(ret);

    printf("open return value ret: %d\n", ret);
    printf("open return value fd: %d\n", fd2);

    // 写文件
    char* pt = "锄禾日当午, 编码真辛苦......";
    write(fd2, pt, strlen(pt));

    close(fd2);

    return 0;
}

    • fcntl

      // 函数原型
// fcntl能根据传参不同,可以完成五件事,这里只介绍其中两件
int fcntl(int fd, int cmd, ... /* arg */ );

// 1. 复制文件描述符
int fcntl(int fd, int cmd, ... /* arg */ );
	参数fd: 要复制的文件描述符
	参数cmd: 控制这个函数到底做什么, 复制文件描述符则使用: F_DUPFD
	返回值: 得到了复制完成之后的新的文件描述符
	
// 举例
int fd  = open("a.txt", O_RDWR);
int newfd = fcntl(fd, F_DUPFD);

      // 2. 修改已经打开的文件的flag属性
//int open(const char *pathname, int flags);
int fcntl(int fd, int cmd, ... /* arg */ );
	参数fd: 打开的文件的文件描述符
	参数cmd: 
    	- 获取文件设置的flags属性: F_GETFL
        - 设置文件的flags属性: F_SETFL
			- 可以被设置的属性: O_APPEND, O_NONBLOCK
			- 不能被设置的属性: O_RDONLY, O_WRONLY,O_RDWR,O_CREAT, O_EXCL,O_TRUNC

      示例

      #include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <fcntl.h>

int main()
{
    int ret = open("abc.txt", O_RDWR|O_CREAT, 0777);
    if(ret == -1){
        perror("open");
    }

    // 追加属性 -> o_append
    // 1. 获取设置的属性
    int fl = fcntl(ret, F_GETFL);
    // 2. 追加属性
    fl = fl | O_APPEND;
    // 3. 将新的属性设置回去
    fcntl(ret, F_SETFL, fl);

    // 写文件
    char* pt = " 追加属性测试....., 是不是写入到文件尾部???";
    write(ret, pt, strlen(pt));

    close(ret);

    return 0;
}

    • makefile

      • 给构建工具make使用的

      • 格式:

        目标 ...:依赖 ...
(tab) 命令
	  ....

      • 规则:

        • 检测依赖是否存在
          • 不存在, 检测其他规则
        • 检测文件是否要更新:
          • 依赖和目标的日期比较
            • 依赖 < 目标 日期
              • 正常情况
            • 依赖 > 目标 日期
              • 不正常
                • 执行当前规则对应的命令, 生成目标
            • 目标不存在
              • 执行当前规则对应的命令, 生成目标

      • 变量:

        • 自定义变量 -> 小写

          • 变量名=变量值
          • 取值: $(变量名)

        • 自带的变量 -> 大写

          • 变量名=变量值
          • 取值: $(变量名)

        • 自动变量

          • 只能在规则中的命令中使用

          • $@: 规则中的目标

          • $<: 规则中的第一个依赖

          • $^: 规则中的所有依赖

            app:a.c b.c c.c
	gcc $^ -o $@

        • 例子

	head.h

	gcc $^ -o $@
	
%.o:%.c

        • 查找指定路径下指定类型的文件

            • 参数: 路径/*.c

            • 参数1: 要替换的字符串 %.c (% 匹配不替换的部分)

            • 参数3: 被替换的原始字符串
